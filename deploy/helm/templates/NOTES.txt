   kind: PersistentVolumeClaim
   metadata:
     name: upcloud-test-pvc
   spec:
     storageClassName: upcloud-block-storage-maxiops
     accessModes:
       - ReadWriteOnce
     resources:
       requests:
         storage: 10Gi
   EOF

4. For snapshot support, create a VolumeSnapshotClass and test:

   kubectl apply -f - <<EOF
   apiVersion: snapshot.storage.k8s.io/v1
   kind: VolumeSnapshot
   metadata:
     name: test-snapshot
   spec:
     volumeSnapshotClassName: {{ .Values.volumeSnapshotClass.name }}
     source:
       persistentVolumeClaimName: upcloud-test-pvc
   EOF
